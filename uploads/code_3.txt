import base64

def encode_binary_file(input_filepath, output_filepath):
    try:
        # Open the input file in binary read mode ('rb')
        with open(input_filepath, "rb") as infile:
            # Read the content of the binary file
            binary_content = infile.read()

        # Encode the binary content using base64
        encoded_content = base64.b64encode(binary_content)

        # Open the output file in write mode to save the encoded content
        with open(output_filepath, "wb") as outfile:
            # Write the base64 encoded content to the output file
            outfile.write(encoded_content)
        print(f"Binary file successfully encoded and saved to {output_filepath}")

    except FileNotFoundError as e:
        print(f"Error: {e}")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")
