    <canvas id="myChart" width="400" height="200"></canvas>
    
    <script>
        // Pass backend data to JavaScript
        const parkingRecords = [
            {% for record in parking_records %}
            {
                vehicleRegNo: "{{ record.vehicle_reg_no }}",
                company: "{{ record.company }}",
                parkingSlot: "{{ record.parking_slot }}",
                amount: {{ record.amount }}
            },
            {% endfor %}
        ];

        // Example of visualizing total income from parking in a bar chart
        const labels = parkingRecords.map(record => record.vehicleRegNo);
        const data = parkingRecords.map(record => record.amount);

        const ctx = document.getElementById('myChart').getContext('2d');
        const myChart = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Parking Fee Amounts',
                    data: data,
                    backgroundColor: 'rgba(54, 162, 235, 0.2)',
                    borderColor: 'rgba(54, 162, 235, 1)',
                    borderWidth: 1
                }]
            },
            options: {
                scales: {
                    y: {
                        beginAtZero: true
                    }
                }
            }
        });
    </script>
